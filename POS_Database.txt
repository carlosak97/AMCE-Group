CREATE TABLE Employee
 (
Emp_ID   Integer,
 
First_Name VARCHAR(50),

Last_Name VARCHAR(50),

DateOfBirth  DATE,
  
PhoneNumber   VARCHAR(9),
  
Position   VARCHAR(50),

Employee_Tag  Integer,

Emp_Pass varchar(20),
 
PRIMARY KEY  (Empe_ID));
 

CREATE TABLE Item(
Item_ID     Integer,

ItemName   VARCHAR(25),
  
ItemSize     VARCHAR(10),
  
ItemColor       VARCHAR(20),
  
Item_Price Integer,

ItemQuantity Integer,

Item_Code Integer,
  
PRIMARY KEY (Item_ID));

   

CREATE TABLE Trans_Foot(
  
Foot_ID Integer,
  
Item Integer,
  
Salesman Integer,
  
Price Integer,
  
Quantity Integer,
  
Total_Price Integer,
  
Total_Quantity Integer, 
  
PRIMARY KEY (Foot_ID),
  
FOREIGN KEY (Item_ID) REFERENCES Item(Item_ID),
  FOREIGN KEY (Salesman) REFERENCES Employee(Employee_ID));
  
 


CREATE TABLE Trans_Head(
  
Tran_ID Integer,
  
Tran_Date Date,
  
Pay_ID Integer,
  
Foot_ID Integer,
  
PRIMARY KEY (Tran_ID),
  
FOREIGN KEY (Pay_ID) REFERENCES Payment(Payment_ID),
  FOREIGN KEY (Foot_ID) REFERENCES Trans_Foot(Foot_ID));
  

CREATE TABLE Customer(

Cust_ID Integer,

Cust_Name varchar(20) ,

Cust_Age char(3) ,

Cust_Telephone  varchar(10),

Cust_Disc Decimal(1, 1), 

Nationality varchar(20),

Region varchar(20),

Cust_DOB Date,

PRIMARY KEY (Cust_ID));
